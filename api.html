get_token

var UserCenter = require('xbm-sdk').UserCenter;
var uc = new UserCenter('13800138000', '10086');

uc.get_token(function (err, token) {
  if (err) {
    console.log(err);
  }

  var token = token;
})

get_user_info

根据 token 获取用户基本信息

Examples:

var UserCenter = require('xbm-sdk').UserCenter;
var uc = new UserCenter('13800138000', '10086');

uc.get_token(function (err, token) {
  if (err) {
    console.log(err);
  }

  var token = token;

  uc.get_user_info(token, function (err, user_info) {
    if (err) {
      console.log(err);
    }

    var user_info = user_info;
  })
})
